Query con SELECT1) SELECT *FROM `students`WHERE YEAR(`date_of_birth`) = '1990'ORDER BY `date_of_birth`;2) SELECT *FROM `courses`WHERE `cfu` > '10'ORDER BY `cfu`;3) SELECT *FROM `students`WHERE ('2022' - YEAR(`date_of_birth`)) > '30'ORDER BY `surname`;4) SELECT *FROM `courses`WHERE `period` = 'I Semestre'AND `year` = '1'ORDER BY `name`;5) SELECT *FROM `exams`WHERE `hour` > '14:00'AND `date` = '2020-06-20'ORDER BY `id`;6) SELECT *FROM `degrees`WHERE `level` = 'magistrale'ORDER BY `name`;7)SELECT COUNT(*) AS `n_dipartimenti`FROM `departments`;8)SELECT COUNT(*) AS `n_insegnanti_no_telefono`FROM `teachers`WHERE `phone` IS NULL;Query con GROUP BY1) SELECT COUNT(*) AS `n_iscritti`, YEAR(`enrolment_date`) AS `anno`FROM `students`GROUP BY `anno`;2) SELECT COUNT(*) AS `n_insegnanti`, `office_address` AS `edificio`FROM `teachers`GROUP BY `edificio`;3)SELECT AVG(`vote`) AS `media_voti`, `exam_id`FROM `exam_student`GROUP BY `exam_id`;4) SELECT COUNT(*) AS `n_corsi`, `department_id`FROM `degrees`GROUP BY `department_id`;
